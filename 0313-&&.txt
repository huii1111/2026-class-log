#include <stdio.h>

int main() {
    int f;

    printf("좋아하는 음식 번호를 선택하세요\n");
    printf("1:치킨 2:피자 3:떡볶이 4:햄버거 5~7: 랜덤\n");

    printf("번호 입력: ");
    scanf("%d", &f);

      if (f==1)
        printf("치킨");
        
      else if (f==2)
        printf("피자");
        
      else if (f==3)
        printf("떡볶이");
        
      else if(f==4)
        printf("햄버거");
        
      else if (f >= 5 && f <= 7) //5이상 7이하의 숫자를 포함해야하니 and의 의미인 && 사용
			   //5 <= f <= 7 이건 안됌	
        printf("랜덤");
            
      else
        printf("입력오류");
    

    return 0;
}
